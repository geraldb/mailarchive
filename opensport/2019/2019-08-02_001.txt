require 'sportdb/config'

## note: requires a local copy of the football.db clubs datasets
##          see https://github.com/openfootball/clubs
SportDb::Import.config.clubs_dir =3D './clubs'

m =3D SportDb::Import.config.clubs.match( 'Arsenal' )
m.size     # 3 club matches found
#=3D> 3
m[0].name; m[0].city; m[0].country
#=3D> "Arsenal FC", "London", "England"
m[1].name; m[1].city; m[1].country
#=3D> "Arsenal Tula", "Tula", "Russia"
m[2].name; m[2].city; m[2].country
#=3D> "Arsenal de Sarand=C3=AD", "Sarand=C3=AD", "Argentina"


m =3D SportDb::Import.config.clubs.match_by( name: 'Arsenal', country: 'eng=
