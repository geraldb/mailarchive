#!/usr/bin/python

from bs4 import BeautifulSoup
import sys

def main():
    def cleanup(s):
        s = s.strip().replace("\t", " ").replace("\n", " ").replace("\r", " ")
        while s.find("  ") >= 0:
            s = s.replace("  ", " ")
        return s
        
    page = sys.stdin.read()
    soup = BeautifulSoup(page)

    h2 = soup.find("h2", sectiontitle="Introduction")
    soup = h2.parent
    start = soup.find("h2", sectiontitle="Introduction")
    print "SECT :", start.text.strip()
    sib = start.find_next()
    while sib:
        if str(sib).startswith('<div class="category"'):
            if str(sib).find('id="field"') > 0:
                print
                print "FIELD:", cleanup(sib.text)
            else:
                txt = cleanup(sib.text)
                col_pos = txt.find(":")
                name = txt[:col_pos]
                print "CAT  :", name
                text = txt[col_pos +1:].strip()
                if text:
                    print "DATA :", text
        elif str(sib).startswith('<div class="category_data"'):
            print "DATA :", cleanup(sib.text)
        elif str(sib).startswith('<h2'):
            print '\n' + '-' * 40 + '\n' 
            print "SECT :", cleanup(sib.text)
        sib = sib.find_next()


if __name__ == "__main__":
    main()
