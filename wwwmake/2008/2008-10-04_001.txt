   Here's an example JavaScript addon for S6 that lets you toggle
(start/stop) autoplay. To add it just create your slide show as usual
(e.g. slideshow kiosk) than open up the slides.js file in the S6
folder and add the autoplay and toogleAutoplay functions:

function autoplay()
{
	// suspend autoplay in outline view (just slideshow view)
	if( !s6mode )
	  return;

	var cid = '#slide' + snum;

	snum += 1;
	if( snum > smax ) snum = 1;

	var nid = '#slide' + snum;

  $( cid ).hide();
  $( nid ).show();

	// update jumplist, slide counter display, and permalink
  $('#jumplist').get(0).selectedIndex = (snum-1);
  currentSlide();
  permaLink();
}

var autoplayId = null;

function toggleAutoplay()
{
  if( autoplayId )
	{
		clearInterval( autoplayId );
		autoplayId = null;
	}
	else
	{
   	autoplayId = setInterval ( autoplay, 2000 );
	}
}

  As a last step pick the keys you want to use for toggling (starting/
stopping) autoplay and add a case to the function keys() switch e.g:

  case 80: //p
  case 83: //s
    toggleAutoplay();
    break;

  That's all there is to it. Let me know if you any questions or if it
works for you.
