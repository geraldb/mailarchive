Here is an even better hack:  the slideshow magically remembers what 
slide you were previously on (using cookies), and when you reload 
the browser, it takes you back there!

<script type="text/javascript">

//
// remember the slide that is currently being viewed
//
$(window).unload(function() {
   document.cookie = "snum=" + snum;
});

//
// restore the slide that was viewed most recently
//
$(document).ready(function() {
   var parts = document.cookie.match( /snum=(\d+)/ );

   if (parts != null && parts.length > 1)
   {
       goTo( parseInt( parts[1] ) );
   }
});

</script>
